namespace Kitab;

use Hoa\File;

/**
 * Custom finder that scans only PHP files.
 */
class Finder extends File\Finder
{
    /**
     * Configure the finder.
     *
     * This finder is pre-configured to scan only PHP files. It ignores Git
     * files and Mercurial files. No dependencies are ignored (for instance
     * Composer).
     */
    public function __construct()
    {
        parent::__construct();

        $this
            ->files()
            ->name('/\.php$/')
            ->notIn('/^\.(git|hg)$/');
    }
}
